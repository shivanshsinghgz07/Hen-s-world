    // Join a specific room

    function joinRoom(roomId) {

        if (!database || !isFirebaseReady) {

            alert("Firebase not ready yet. Please wait or refresh the page.");

            return false;

        }

       

        // Clean room ID (no special path issues)

        const cleanRoomId = roomId.trim().replace(/[.#$[\]]/g, '_');

        if (!cleanRoomId) return false;

       

        if (currentRoom === cleanRoomId) {

            // Already in this room

            return true;

        }

       

        // Leave old room listeners by nullifying reference (onChildAdded will be detached)

        if (messagesRef) {

            messagesRef = null;

        }

       

        currentRoom = cleanRoomId;

        document.getElementById("currentRoomName").innerText = currentRoom;

       

        // Enable message input

        const msgInput = document.getElementById("messageInput");

        const sendBtn = document.getElementById("sendBtn");

        msgInput.disabled = false;

        sendBtn.disabled = false;

        msgInput.focus();

       

        // Clear messages container and show loading

        const container = document.getElementById("messagesContainer");

        container.innerHTML = '<div class="loading">🐔 joining the coop... loading messages</div>';

       

        // Build reference to messages in this room

        messagesRef = ref(database, `rooms/${currentRoom}/messages`);

        const messagesQuery = query(messagesRef, limitToLast(100));

       

        // Listen for new/old messages

        onChildAdded(messagesQuery, (snapshot) => {

            const message = snapshot.val();

            if (message) {

                displayMessage(message);

            }

        });

       

        // Add system join message after a short delay (avoid race)

        setTimeout(() => {

            if (currentRoom === cleanRoomId) {

                addSystemMessage(`${username} 🐔 joined the coop`);

            }

        }, 300);

       

        return true;

    }

   

    // Send a text message to current room

    async function sendMessage(text) {

        if (!text.trim()) return false;

        if (!currentRoom) {

            alert("Join a room first!");

            return false;

        }

        if (!database || !isFirebaseReady) {

            alert("Waiting for Firebase connection...");

            return false;

        }

       

        const messageData = {

            sender: username,

            text: text.trim(),

            timestamp: serverTimestamp(),

            timestampMs: Date.now(),

            type: "user"

        };

       

        try {

            await push(ref(database, `rooms/${currentRoom}/messages`), messageData);

            // Clear input after successful send

            document.getElementById("messageInput").value = "";

            return true;

        } catch (error) {

            console.error("Send error:", error);

            alert("Failed to send message. Check network/Firebase rules.");

            return false;

        }

    }

   

    // Add system notification (like join/leave events)

    async function addSystemMessage(text) {

        if (!currentRoom || !database || !isFirebaseReady) return;

       

        const systemMsg = {

            sender: "🐔 Hen's World",

            text: text,

            timestamp: serverTimestamp(),

            timestampMs: Date.now(),

            isSystem: true,

            type: "system"

        };

       

        try {

            await push(ref(database, `rooms/${currentRoom}/messages`), systemMsg);

        } catch (error) {

            console.warn("System message not sent:", error);

        }

    }

   

    // Escape HTML to prevent injection

    function escapeHtml(str) {

        if (!str) return "";

        const div = document.createElement('div');

        div.textContent = str;

        return div.innerHTML;

    }

   

    // Display message in chat container

    function displayMessage(message) {

        const container = document.getElementById("messagesContainer");

        if (!container) return;

       

        // Remove placeholder loading if exists

        if (container.children.length === 1 && container.children[0].classList?.contains('loading')) {

            container.innerHTML = '';

        }

       

        const msgDiv = document.createElement("div");

        const isOwn = (message.sender === username);

        const isSystem = message.isSystem === true || message.type === "system" || message.sender === "🐔 Hen's World";

       

        if (isSystem) {

            msgDiv.className = "message system-message";

        } else {

            msgDiv.className = `message ${isOwn ? 'own-message' : ''}`;

        }

       

        let timeStr = "now";

        if (message.timestampMs) {

            const date = new Date(message.timestampMs);

            timeStr = date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', hour12: false });

        } else if (message.timestamp && typeof message.timestamp === 'object') {

            // fallback for serverTimestamp placeholder

            timeStr = "just now";

        }

       

        const senderDisplay = isSystem ? "🐔 system" : escapeHtml(message.sender);

       

        msgDiv.innerHTML = `

            <div class="message-header">

                <span class="message-sender">${senderDisplay}</span>

                <span class="message-time">${timeStr}</span>

            </div>

            <div class="message-text">${escapeHtml(message.text)}</div>

        `;

       

        container.appendChild(msgDiv);

        container.scrollTop = container.scrollHeight;

    }

    }

        const statusDiv = document.getElementById("connectionStatus");

        if (statusDiv) {

            let connText = isFirebaseReady ? "✅ live" : "⏳ connecting...";

            statusDiv.innerHTML = `👤 ${escapeHtml(username)} · ${connText}`;

        }

    }
